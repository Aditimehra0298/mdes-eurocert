===========================================

Copy this SIMPLE test script into your Apps Script:

function testSheetAccess() {
  // Test 1: Can we access the spreadsheet?
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  Logger.log("‚úÖ Spreadsheet Name: " + ss.getName());
  
  // Test 2: List all sheet tabs
  const sheets = ss.getSheets();
  Logger.log("üìã Found " + sheets.length + " sheets:");
  sheets.forEach(sheet => {
    Logger.log("  - " + sheet.getName());
  });
  
  // Test 3: Try to find "Consultation Bookings" sheet
  const consultationSheet = ss.getSheetByName('Consultation Bookings');
  if (consultationSheet) {
    Logger.log("‚úÖ Found 'Consultation Bookings' sheet");
    
    // Test 4: Try to write a test row
    consultationSheet.appendRow([
      new Date(),
      "TEST USER",
      "test@email.com",
      "+919999999999",
      "TEST COMPANY",
      "This is a test from Apps Script"
    ]);
    Logger.log("‚úÖ Test row written successfully!");
    
  } else {
    Logger.log("‚ùå ERROR: Cannot find 'Consultation Bookings' sheet");
    Logger.log("‚ö†Ô∏è  Sheet names are case-sensitive!");
    Logger.log("‚ö†Ô∏è  Make sure tab is named exactly: 'Consultation Bookings'");
  }
  
  // Test 5: Try to find "Event Registrations" sheet
  const eventSheet = ss.getSheetByName('Event Registrations');
  if (eventSheet) {
    Logger.log("‚úÖ Found 'Event Registrations' sheet");
  } else {
    Logger.log("‚ùå ERROR: Cannot find 'Event Registrations' sheet");
  }
}
